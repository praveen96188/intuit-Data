
CREATE TRIGGER iqclientInsertPSP
 AFTER  INSERT ON iqclient
 REFERENCING NEW AS nRow
 FOR EACH ROW MODE DB2ROW
 WHEN (
  nRow.CLI_STATUS IS NOT NULL AND TRIM(nRow.CLI_STATUS) <> ''
 )
 BEGIN ATOMIC
    INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID) VALUES (nRow.CLI_USERID, CURRENT TIMESTAMP,  'IQCLIENT', NULL);
 END


CREATE TRIGGER iqclientUpdatePSP
AFTER  UPDATE ON iqclient
REFERENCING NEW AS nRow OLD AS oRow
FOR EACH ROW MODE DB2ROW
WHEN (
  nRow.CLI_USERID <> oRow.CLI_USERID OR
  nRow.CLI_FEIN <> oRow.CLI_FEIN OR
  nRow.CLI_ORIGINAL_FEIN <> oRow.CLI_ORIGINAL_FEIN OR
  nRow.CLI_LEGALNAME <> oRow.CLI_LEGALNAME OR
  nRow.CLI_DBA <> oRow.CLI_DBA OR
  nRow.CLI_ADDR1 <> oRow.CLI_ADDR1 OR
  nRow.CLI_ADDR2 <> oRow.CLI_ADDR2 OR
  nRow.CLI_CITY <> oRow.CLI_CITY OR
  nRow.CLI_STATE <> oRow.CLI_STATE OR
  nRow.CLI_POSTALCODE <> oRow.CLI_POSTALCODE OR
  nRow.CLI_MAIL_ADDR1 <> oRow.CLI_MAIL_ADDR1 OR
  nRow.CLI_MAIL_ADDR2 <> oRow.CLI_MAIL_ADDR2 OR
  nRow.CLI_MAIL_CITY <> oRow.CLI_MAIL_CITY OR
  nRow.CLI_MAIL_STATE <> oRow.CLI_MAIL_STATE OR
  nRow.CLI_MAIL_POSTALCODE <> oRow.CLI_MAIL_POSTALCODE OR
  nRow.CLI_REGNUM <> oRow.CLI_REGNUM OR
  nRow.CLI_APPVER <> oRow.CLI_APPVER OR
  nRow.CLI_APPID <> oRow.CLI_APPID OR
  nRow.CLI_AGREE_NUM <> oRow.CLI_AGREE_NUM OR
  (nRow.CLI_STATUS IS NOT NULL AND TRIM(nRow.CLI_STATUS) <> '' AND nRow.CLI_STATUS <> oRow.CLI_STATUS) OR
  nRow.CLI_ACTIV_STATUS <> oRow.CLI_ACTIV_STATUS OR
  nRow.CLI_ON_HOLD <> oRow.CLI_ON_HOLD OR
  nRow.CLI_APPKEY <> oRow.CLI_APPKEY OR
  nRow.CLI_LASTTAXQTR <> oRow.CLI_LASTTAXQTR
  )
BEGIN ATOMIC
    INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID) VALUES (nRow.CLI_USERID, CURRENT TIMESTAMP,  'IQCLIENT', NULL);
    IF (nRow.CLI_ORIGINAL_FEIN <> oRow.CLI_ORIGINAL_FEIN) THEN
                    INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID) VALUES (nRow.CLI_USERID, CURRENT TIMESTAMP,  'ENTCHANGE', NULL);
    END IF;
END

CREATE TRIGGER iqpitemUpdatePSP
AFTER  UPDATE ON iqpitem
REFERENCING NEW AS nRow OLD AS oRow
 FOR EACH ROW MODE DB2ROW
WHEN (nRow.PIT_PITEMTYPE='T' and (nRow.PIT_COMPID <> oRow.PIT_COMPID OR
nRow.PIT_RATE <> oRow.PIT_RATE OR
nRow.PIT_RATE_PERIODS <> oRow.PIT_RATE_PERIODS))
BEGIN ATOMIC
    DECLARE USERID VARCHAR(128);
    SELECT SESSION_USER INTO USERID FROM SYSIBM/SYSDUMMY1;
    IF USERID <> 'QPGMR' THEN
        INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID) 
        VALUES( nRow.PIT_USERID, CURRENT TIMESTAMP, 'IQPITEM', nRow.PIT_PITEMID);
    END IF;
END

CREATE TRIGGER iqpitemaddUpdatePSP
AFTER  UPDATE ON iqpitemadd
REFERENCING NEW AS nRow OLD AS oRow
 FOR EACH ROW MODE DB2ROW
 WHEN (nRow.PITA_INACTIVE <> oRow.PITA_INACTIVE OR nRow.PITA_EXEMPT <> oRow.PITA_EXEMPT OR nRow.PITA_REIMBURSE <> oRow.PITA_REIMBURSE)
BEGIN ATOMIC
    DECLARE USERID VARCHAR(128);
    DECLARE PITEMTYPE VARCHAR(128);
    SELECT PIT_PITEMTYPE into PITEMTYPE from IQPITEM where PIT_USERID=nRow.PITA_USERID and PIT_PITEMID=nRow.PITA_PITEMID;
    SELECT SESSION_USER INTO USERID FROM SYSIBM/SYSDUMMY1;
    IF PITEMTYPE = 'T' AND USERID <> 'QPGMR' THEN
        INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID)
        VALUES( nRow.PITA_USERID, CURRENT TIMESTAMP, 'IQPITEM', nRow.PITA_PITEMID);
    END IF;
END

CREATE TRIGGER txaciInsertPSP                                       
AFTER  INSERT ON txaci                                      
REFERENCING NEW AS nRow 
FOR EACH ROW MODE DB2ROW                           
WHEN (
	nRow.ACI_CODE='STDEPFRQ' OR
	(nRow.ACI_CODE = 'EINCHANGE' AND nRow.ACI_STATE = 'FD')
	OR nRow.ACI_CODE in ('BUSTAXID','BANKACCT','ACHWHPIN','EFTNBR', 'REGACHW', 'REGACHU', 'SUICREDIT', 'UHICREDIT', 'WHCREDIT', 'UIAS', 'UIHIRATE')
)                           
BEGIN ATOMIC
DECLARE USERID VARCHAR(128);
DECLARE PSID DECIMAL(9);
SELECT SESSION_USER INTO USERID FROM SYSIBM/SYSDUMMY1;
SELECT CLI_USERID INTO PSID 
FROM IQCLIENT 
WHERE CLI_FEIN = CONCAT(SUBSTR(nRow.ACI_EIN, 1, 2),SUBSTR(nRow.ACI_EIN, 4,7))
AND CLI_STATUS NOT IN ('CC', 'TI');  
IF USERID <> 'QPGMR' THEN
	IF nRow.ACI_CODE='STDEPFRQ' THEN
		INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID) 
		VALUES( CONCAT(SUBSTR(nRow.ACI_EIN, 1, 2), SUBSTR(nRow.ACI_EIN, 4, 7)),
				CURRENT TIMESTAMP,
				'TXACI',
				nRow.ACI_STATE);
	END IF;
END IF;
IF nRow.ACI_CODE = 'EINCHANGE' THEN
    IF (PSID is not null) THEN
                    INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID) VALUES (PSID, CURRENT TIMESTAMP, 'ENTCHANGE', NULL);
    END IF;
END IF;
IF nRow.ACI_CODE IN ('BUSTAXID','BANKACCT','ACHWHPIN','EFTNBR') THEN
    INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID) VALUES (CONCAT(SUBSTR(nRow.ACI_EIN, 1, 2), SUBSTR(nRow.ACI_EIN, 4, 7)), CURRENT TIMESTAMP, 'TXACIAID', NULL);
END IF;
IF nRow.ACI_CODE IN ('REGACHW', 'REGACHU') THEN
    INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID) VALUES (CONCAT(SUBSTR(nRow.ACI_EIN, 1, 2), SUBSTR(nRow.ACI_EIN, 4, 7)), CURRENT TIMESTAMP, nRow.ACI_CODE, NULL);
END IF;
IF nRow.ACI_CODE in ('SUICREDIT', 'UHICREDIT', 'WHCREDIT', 'UIAS', 'UIHIRATE') THEN
    INSERT INTO PSPEVENT (USERID, TIMESTAMP, SOURCE, SOURCEID) VALUES (CONCAT(SUBSTR(nRow.ACI_EIN, 1, 2), SUBSTR(nRow.ACI_EIN, 4, 7)), CURRENT TIMESTAMP, nRow.ACI_CODE, nRow.ACI_STATE);
END IF;
END
